Exercise 3.2: Fine-tuning a CLIP Model (harder)¶

Use a (small) CLIP model like openai/clip-vit-base-patch16 and evaluate its zero-shot performance on a small image classification dataset like ImageNette or TinyImageNet. Fine-tune (using a parameter-efficient method!) the CLIP model to see how much improvement you can squeeze out of it.

Note: There are several ways to adapt the CLIP model; you could fine-tune the image encoder, the text encoder, or both. Or, you could experiment with prompt learning.

Tip: CLIP probably already works very well on ImageNet and ImageNet-like images. For extra fun, look for an image classification dataset with different image types (e.g. sketches).

import torch
from torch.utils.data import DataLoader

# ==========================================
# 1. FUNZIONE DI COLLAZIONE
# ==========================================
def val_collate_fn(batch):
    images, labels = zip(*batch)
    return list(images), torch.tensor(labels)

# ==========================================
# 2. ESTRAZIONE FEATURE CORRETTA
# ==========================================
def get_image_features_safe(model, image_inputs):
    """
    Dopo il wrapping PEFT, vision_model restituisce BaseModelOutputWithPooling.
    Estraiamo le feature corrette e le normalizziamo.
    """
    vision_outputs = model.vision_model(**image_inputs)
    # pooler_output è il [CLS] token proiettato, shape: (batch, 512)
    image_features = vision_outputs.pooler_output
    # Proiezione nello spazio condiviso CLIP
    image_features = model.visual_projection(image_features)
    # Normalizzazione L2
    image_features = image_features / image_features.norm(dim=-1, keepdim=True)
    return image_features

def get_text_features_safe(model, text_inputs):
    """
    Estrazione feature testuali robusta e normalizzata.
    """
    text_outputs = model.text_model(**text_inputs)
    # pooler_output per il testo: embedding del token [EOS]
    text_features = text_outputs.pooler_output
    # Proiezione nello spazio condiviso CLIP
    text_features = model.text_projection(text_features)
    # Normalizzazione L2
    text_features = text_features / text_features.norm(dim=-1, keepdim=True)
    return text_features

# ==========================================
# 3. LOGIT SCALE FISSATO
# ==========================================
# Il logit_scale è esploso dopo il wrapping PEFT.
# Il valore corretto di CLIP originale è log(1/0.07) ≈ 2.659 → exp ≈ 14.3
# Lo resettiamo al valore originale prima della valutazione.
with torch.no_grad():
    model.logit_scale.fill_(torch.tensor(2.6592))  # log(1/0.07)

print(f"logit_scale dopo reset: {model.logit_scale.exp().item():.4f}")  # atteso ~14.3

# ==========================================
# 4. FUNZIONE DI VALUTAZIONE
# ==========================================
def evaluate_zeroshot(loader, model, processor, class_names, device):
    model.eval()
    correct = 0
    total = 0

    # Prompt engineering
    prompts = [f"a photo of a {cls}" for cls in class_names]
    text_inputs = processor(text=prompts, return_tensors="pt", padding=True).to(device)

    # Estrazione feature testuali (classificatore sintetizzato, fatto una volta sola)
    with torch.no_grad():
        text_features = get_text_features_safe(model, text_inputs)

    # Loop sulle immagini
    for images, labels in loader:
        # Solo pixel_values serve per il vision_model
        image_inputs = processor(images=images, return_tensors="pt")
        pixel_values = image_inputs["pixel_values"].to(device)

        with torch.no_grad():
            image_features = get_image_features_safe(
                model, {"pixel_values": pixel_values}
            )

            # Similarità coseno scalata
            logits = image_features @ text_features.T * model.logit_scale.exp()
            preds = logits.argmax(dim=-1).cpu().numpy()

        correct += (preds == labels.numpy()).sum()
        total += len(labels)

    return correct / total

# ==========================================
# 5. ESECUZIONE BASELINE
# ==========================================
val_loader = DataLoader(
    val_ds_tv,
    batch_size=32,
    shuffle=False,
    collate_fn=val_collate_fn
)

print("Calcolo Zero-Shot Baseline...")
zeroshot_acc = evaluate_zeroshot(val_loader, model, processor, class_names, device)
print(f"\n✅ Zero-Shot Baseline Accuracy: {zeroshot_acc * 100:.2f}%")

import torch
from torch.utils.data import DataLoader, TensorDataset
from torch.amp import autocast, GradScaler
from transformers import get_cosine_schedule_with_warmup
from tqdm import tqdm

# ==========================================
# 1. PRE-ELABORAZIONE IN RAM
# ==========================================
# Carichiamo il dataset torchvision "nudo" (PIL Images)
train_ds_raw = torchvision.datasets.Imagenette(root="./data", split="train", size="320px", download=True, transform=None)

class CLIPFastDataset(torch.utils.data.Dataset):
    def __init__(self, dataset, class_names, processor):
        print("Pre-elaborazione del dataset in corso... (Attendere circa 2-3 minuti)")
        self.items = []
        for i in tqdm(range(len(dataset))):
            image, label = dataset[i]
            prompt = f"a photo of a {class_names[label]}"
            # Il processor viene chiamato qui UNA VOLTA sola per tutto il dataset
            inputs = processor(text=prompt, images=image, return_tensors="pt", padding="max_length", truncation=True)
            self.items.append({
                "pixel_values": inputs["pixel_values"].squeeze(0),
                "input_ids": inputs["input_ids"].squeeze(0),
                "attention_mask": inputs["attention_mask"].squeeze(0)
            })

# Qui viene creata la variabile che ti dava l'errore!
train_fast_ds = CLIPFastDataset(train_ds_raw, class_names, processor)

# ==========================================
# 2. DATASET OTTIMIZZATO
# ==========================================
print("Conversione in tensori monolitici per eliminare l'overhead Python...")
pixel_values = torch.stack([item["pixel_values"] for item in train_fast_ds.items])
input_ids = torch.stack([item["input_ids"] for item in train_fast_ds.items])
attention_mask = torch.stack([item["attention_mask"] for item in train_fast_ds.items])

# TensorDataset è infinitamente più veloce su Windows
train_final_ds = TensorDataset(pixel_values, input_ids, attention_mask)

train_loader = DataLoader(
    train_final_ds, 
    batch_size=32, 
    shuffle=True, 
    num_workers=0, # Con TensorDataset, 0 è stabilissimo su Windows
    pin_memory=True
)

# ==========================================
# 3. OTTIMIZZAZIONE MODELLO (Specifiche OpenAI CLIP)
# ==========================================
# Abilitiamo il gradient checkpointing per risparmiare VRAM e velocizzare 
model.gradient_checkpointing_enable()

# Iperparametri Adam specifici per CLIP ViT: beta2=0.98, eps=1e-6
optimizer = torch.optim.AdamW(    
    filter(lambda p: p.requires_grad, model.parameters()),
    lr=5e-5,
    betas=(0.9, 0.98),
    eps=1e-6,
    weight_decay=0.2
)

# ==========================================
# 0. DEFINIZIONE IPERPARAMETRI
# ==========================================
EPOCHS = 3  # Definiamo la variabile mancante
BATCH_SIZE = 32

scaler = GradScaler('cuda')
scheduler = get_cosine_schedule_with_warmup(
    optimizer,
    num_warmup_steps=100,
    num_training_steps=EPOCHS * len(train_loader)
)

# ==========================================
# 4. TRAINING LOOP ACCELERATO (Mixed Precision)
# ==========================================
model.train()
print(f"\nInizio training accelerato su {device}...")

for epoch in range(EPOCHS):
    total_loss = 0
    pbar = tqdm(train_loader, desc=f"Epoch {epoch+1}/{EPOCHS}")
    
    for batch in pbar:
        # Unpacking veloce dal TensorDataset
        pv, ids, am = [v.to(device, non_blocking=True) for v in batch]
        
        optimizer.zero_grad(set_to_none=True) # Più efficiente [7]

        with autocast('cuda'): # [8]
            outputs = model(pixel_values=pv, input_ids=ids, attention_mask=am, return_loss=True)
            loss = outputs.loss

        scaler.scale(loss).backward()
        scaler.step(optimizer)
        scaler.update()
        
        scheduler.step()
        total_loss += loss.item()
        pbar.set_postfix({"loss": f"{loss.item():.4f}"})

    print(f"✅ Epoca {epoch+1} completata!")

import torch
from torch.utils.data import DataLoader

# ==========================================
# 1. DEFINIZIONE FUNZIONE DI COLLAZIONE
# ==========================================
def val_collate_fn(batch):
    """
    Gestisce il batch di ImageNette (PIL Image, Label) evitando 
    l'errore di collazione dei dati non-tensor [1, 2].
    """
    images, labels = zip(*batch)
    # Restituiamo le immagini come lista (formato accettato dal processor) 
    # e le etichette come tensore.
    return list(images), torch.tensor(labels)

# ==========================================
# 2. CONFIGURAZIONE DATALOADER E VALUTAZIONE
# ==========================================
print("Calcolo accuratezza post-training LoRA...")

# Applichiamo la collate_fn per evitare il TypeError sulle immagini PIL
val_loader = DataLoader(
    val_ds_tv, 
    batch_size=32, 
    shuffle=False, 
    collate_fn=val_collate_fn, # <--- Risolve l'errore
    num_workers=0 # Mantieni 0 per stabilità su Windows [3, 4]
)

# Assicuriamoci che il modello sia in modalità valutazione
model.eval()

# Eseguiamo la valutazione (la tua funzione evaluate_zeroshot ora riceverà dati corretti)
final_acc = evaluate_zeroshot(val_loader)

# ==========================================
# 3. REPORT FINALE DEI RISULTATI
# ==========================================
print(f"\n--- Risultati Finali Esercizio 3.2 ---")
print(f"✅ Accuratezza Zero-Shot Iniziale: {zeroshot_acc*100:.2f}%")
print(f"✅ Accuratezza Post-Fine-Tuning (LoRA): {final_acc*100:.2f}%")

delta = (final_acc - zeroshot_acc) * 100
